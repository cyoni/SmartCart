                    System.out.println(dataSnapshot.getChildrenCount() + "= num");
                    DataSnapshot tmp;

                    while (a.hasNext()) {
                        tmp = a.next();

                        String n = dataMap.get("price")+"";

                        System.out.println("name: " + x + ", price: " + n);

                }


            }
